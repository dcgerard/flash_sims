Time Comparisons

David Gerard

2016-03-30

Abstract

Just a rough comparison of times for implementation. flash runs about twice as fast as tflash.

Rank 1 times

I generate some rank 1 square matrix data of varying dimension sizes and calculate how much time tflash and flash take to compute. I forced both to run exactly 100 iterations.

library(flashr)
set.seed(141)
nseq <- c(10, 100, 1000)
pseq <- c(10, 100, 1000)

time_mat <- matrix(NA, nrow = length(nseq), ncol = 2)
for(index in 1:length(nseq)) {
  n <- nseq[index]
  p <- nseq[index]
  
  l <- rnorm(n)
  f <- rnorm(p)
  Theta <- l %*% t(f)
  E <- matrix(rnorm(n * p), nrow = n)
  Y <- Theta + E  
  
  if(index == 1) {
    tflash(Y, itermax = 2, tol = -1)
    flashr:::flash(Y, maxiter_r1 = 2, tol = -1)
  }
  
  tstart <- proc.time()
  tout <- tflash_homo(Y, itermax = 100, tol = -1, sig_start_itermax = 2)
  ttot <- proc.time() - tstart
  
  fstart <- proc.time()
  fout <- flashr:::flash(Y, maxiter_r1 = 100, tol = -1)
  ftot <- proc.time() - fstart
  
  time_mat[index, 1] <- ttot[3]
  time_mat[index, 2] <- ftot[3]
}
library(xtable)
library(ggplot2)
colnames(time_mat) <- c("Tensor", "Matrix Only")
rownames(time_mat) <- nseq
print(xtable(time_mat, caption = "Time to run in seconds. Rows contain the
             dimensions of the square matrix."), type = "html")

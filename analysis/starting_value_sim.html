Posterior Estimates Similar to Empirical Singular Vectors

David Gerard

2016-04-07

Abstract

Again, I look at the relationship between flash and the empirical singular vectors.

Same Thing on Simulated Data

I generate data with a rank 2 mean.

library(flashr)
library(ggplot2)
rm(list = ls())
set.seed(810)
n <- 100
p <- 90
l <- matrix(rnorm(n * 2), nrow = n)
f <- matrix(rnorm(p * 2), nrow = p)
Theta <- l %*% t(f)
E <- matrix(rnorm(n * p), nrow = n, ncol = p)
Y <- Theta + E

tout <- tflash(Y)
fout <- flash(Y)

tl_scale <- scale(tout$post_mean[[1]] * sign(tout$post_mean[[1]][1]), center = FALSE)
tf_scale <- scale(tout$post_mean[[2]] * sign(tout$post_mean[[2]][1]), center = FALSE)
ff_scale <- scale(fout$f * sign(fout$f[1]), center = FALSE)
fl_scale <- scale(fout$l * sign(fout$l[1]), center = FALSE)

initl_scale <- scale(tout$init_return$ex_list[[1]] *
                     sign(tout$init_return$ex_list[[1]][1]), center = FALSE)
initf_scale <- scale(tout$init_return$ex_list[[2]] *
                     sign(tout$init_return$ex_list[[2]][1]), center = FALSE)

qplot(initl_scale,

The estimates are only slightly different. And the main differences are near zero.

plot(l[, 1], tout$post_mean[[1]],

legend("topright", col = c(1,2), c("Posterior Mean", "Empricial SV"), pch = c(1, 1))

Same thing on the residuals

resid <- Y - fout$l %*% t(fout$f)
tout_resid <- tflash(resid)
qplot(tout_resid$init_return$ex_list[[1]], tout_resid$post_mean[[1]],
      xlab = "Empirical SV", ylab = "Posterior Mean",
      main = "Empirical SV vs Post Mean")

plot(l[, 2], tout_resid$post_mean[[1]],

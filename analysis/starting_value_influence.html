Posterior Estimates Same as Starting Values

David Gerard

2016-04-07

Abstract

This analysis acts as a sort-of justification for using the first few singular vectors as starting values. The posterior means are all close to the emprical singular vectors except for the components near zero, which are shrunk toward zero. When random starting starting directions are used, I only ever see either (1) all of the components are shrunk to zero or (2) we see the posterior means are close to the empirical singular vectors.

Load in Cleaned and Normalized GTEX Data

I load in the cleaned and normalized GTEX data from first_analysis.html and extract the first 100 genes and first 100 individuals for the adipose tissue. Thus we have a 100 by 100 matrix.

library(flashr)
library(ggplot2)
data_path <- "~/Data/gtex/v6_gene_reads/"
load(paste0(data_path, "gtex_array_1000.Rdata"))

small_gtex_mat <- gtex_array[1:100, 1:100, 1]

For the adipose tissue, there are very few missing values. Only 0.09 of the observations are missing.

Fit FLASH and compare to starting values.

tflash is now able to output the starting values. I compare the starting values of the components to the final estimates. flash and tflash use the same starting values. The empirical singular vectors and the posterior means are almost identical, except for components near zero.

tout <- tflash(Y = small_gtex_mat)
fout <- suppressMessages(flash(Y = small_gtex_mat))

tl_scale <- scale(tout$post_mean[[1]] * sign(tout$post_mean[[1]][1]), center = FALSE)
tf_scale <- scale(tout$post_mean[[2]] * sign(tout$post_mean[[2]][1]), center = FALSE)
ff_scale <- scale(fout$f * sign(fout$f[1]), center = FALSE)
fl_scale <- scale(fout$l * sign(fout$l[1]), center = FALSE)
initl_scale <- scale(tout$init_return$ex_list[[1]], center = FALSE)
initf_scale <- scale(tout$init_return$ex_list[[2]], center = FALSE)

qplot(initl_scale,
      tl_scale,
      xlab = "Starting Loadings", ylab = "Final Loadings",
      main = "Start vs Final") +
    geom_abline(slope = 1, intercept = 0)

qplot(initf_scale,
      tf_scale,
      xlab = "Starting Factors", ylab = "Final Factors",
      main = "Start vs Final") +
    geom_abline(slope = 1, intercept = 0)

qplot(scale(tout$init_return$ex_list[[1]], center = FALSE),
      fl_scale,
      xlab = "Starting Loadings", ylab = "Final Loadings from FLASH",
      main = "Start vs Final Using FLASH") +
    geom_abline(slope = 1, intercept = 0)

qplot(scale(tout$init_return$ex_list[[2]], center = FALSE),

Greedy FLASH

Now I do the same thing but using the residuals from the previous iteration, as in greedy FLASH. We see the same thing as above.

resid <- small_gtex_mat

for(index in 1:10) {
    resid <- resid - fout$l %*% t(fout$f)
    tout <- tflash(Y = resid)
    fout <- suppressMessages(flash(Y = resid))

    if(all(abs(tout$post_mean[[1]]) < 10^-6)) {
        break
    }
    if(all(abs(fout$l) < 10^-6)) {
        break
    }

    tl_scale <- scale(tout$post_mean[[1]] * sign(tout$post_mean[[1]][1]), center = FALSE)
    tf_scale <- scale(tout$post_mean[[2]] * sign(tout$post_mean[[2]][1]), center = FALSE)
    ff_scale <- scale(fout$f * sign(fout$f[1]), center = FALSE)
    fl_scale <- scale(fout$l * sign(fout$l[1]), center = FALSE)

    initl_scale <- scale(tout$init_return$ex_list[[1]] *
                         sign(tout$init_return$ex_list[[1]][1]),
                         center = FALSE)
    initf_scale <- scale(tout$init_return$ex_list[[2]] *
                         sign(tout$init_return$ex_list[[2]][1]),
                         center = FALSE)
    
    t1 <- qplot(initl_scale,
                tl_scale,
                xlab = "Starting Loadings", ylab = "Final Loadings",
                main = paste0("Start vs Final, Iter: ", index)) +
        geom_abline(slope = 1, intercept = 0)
    
    t2 <- qplot(initf_scale,
                tf_scale,
                xlab = "Starting Factors", ylab = "Final Factors",
                main = paste0("Start vs Final, Iter: ", index)) +
        geom_abline(slope = 1, intercept = 0)
    
    t3 <- qplot(initl_scale,
                fl_scale,
                xlab = "Starting Loadings", ylab = "Final Loadings from FLASH",
                main = paste0("Start vs Final FLASH, Iter: ", index)) +
        geom_abline(slope = 1, intercept = 0)
    
    t4 <- qplot(initf_scale,
                ff_scale,
                xlab = "Starting Factors", ylab = "Final Factors from FLASH",
                main = paste0("Start vs Final FLASH, Iter: ", index)) +
        geom_abline(slope = 1, intercept = 0)

    cat("Iteration:", index, "\n")
    print(t1)
    print(t2)
    print(t3)
    print(t4)
}

Random Starting Directions

Whenever I start in random directions, the posterior estimates either shrink to zero to move to the empirical singular vectors. Indicating that there dont seem to be any other local modes.

set.seed(818)
for(index in 1:20) {
    tout_rand <- tflash(Y, start = "random")

    ## resid <- Y - tout_rand$post_mean[[1]] %*% t(tout_rand$post_mean[[2]])
    ## flashr:::obj(n, p, resid, tout_rand$sigma_est, tout_rand$post_mean[[1]],
    ##              tout_rand$post_mean[[2]], "lowerbound_lik")
    
    t1 <- qplot(tout_rand$init_return$ex_list[[1]],
                tout_rand$post_mean[[1]],
                xlab = "Starting Loadings", ylab = "Final Loadings",
                main = "Start vs Final")
    
    t2 <- qplot(tout_rand$init_return$ex_list[[2]],
                tout_rand$post_mean[[2]],
                xlab = "Starting Factors", ylab = "Final Factors",
                main = "Start vs Final")

    t3 <- qplot(tout_rand$post_mean[[1]], tout$init_return$ex_list[[1]],
                ylab = "Empirical SV", xlab = "Posterior Means",
                main = "Post Mean vs Empirical SV")
    
    t4 <- qplot(tout_rand$post_mean[[2]], tout$init_return$ex_list[[2]],
                ylab = "Empirical SV", xlab = "Posterior Means",
                main = "Post Mean vs Empirical SV")

    print(t4)
